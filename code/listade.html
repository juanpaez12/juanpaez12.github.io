                @Override
                public boolean remove(Object object) {
                    try{
                        if(object != null){
                            boolean eliminado = false;
                            ListNodeDouble node = head;
                            while(eliminado == false && node != null){
                                if(node.getObject() == object){
                                    this.size--;
                                    eliminado = true;
                                    if(node.previous == null){
                                        head = head.next;
                                        head.previous = null;
                                        
                                    }else{
                                        node.previous.next = node.next;
                                    
                                    }
                                }
                                node = node.next;
                            }
                            return eliminado;
                        }
                        
                        return false;
                    }catch(Exception e){
                        return false;
                    }
                }
