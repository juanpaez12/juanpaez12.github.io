                public boolean remove(Object object) { // 
                    try{
                        ListNode a = head;
                        ListNode anterior = tail;
                        int t = 0;
                        boolean eliminado = false;
                        while(t < size && eliminado == false){
                            t++;
                            if( a.getObject() == object){
                                eliminado = true;
                        this.size--;
                                if(anterior == tail){
                                    head = a.next;
                                    tail.next = head;
                                }else{
                                    anterior.next = a.next;
                                }
                            }
                            anterior = a;
                            a = a.next;
                        }
                        return eliminado;
                    } catch(Exception e){
                        return false;
                    }
                }

                public Iterator iterator() {
                    inode = head;
                    return new Iterator() {
                        @Override
                        public boolean hasNext() {
                            return inode.next != null;
                        }

                        @Override
                        public ListNode next() {
                            if (inode != null) {
                                ListNode tmp = inode;
                                inode = inode.next;
                                return tmp;
                            } else {
                                return null;
                            }
                        }
                    };
                }
