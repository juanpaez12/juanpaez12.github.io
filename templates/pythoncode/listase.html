                from ListInterface import ListInterface
                from ListNode import ListNode
                import numpy as np
                
                class List(ListInterface):
                    inode = ListNode()
                    size = 0
                    head = ListNode()
                    tail = ListNode()
                    
                    def __init__(self):
                        self.clear()
                
                    @classmethod
                    def List(cls,objecto):
                        cls.add(objecto)
                    
                    def isEmpty(self):
                        return self.head == None
                    
                    def getSize(self):
                        return self.size
                    
                    def clear(self):
                        self.head = None
                        self.tail = None
                        self.size = 0
                    
                    def getHead(self):
                        return self.head
                    
                    def getTail(self):
                        return self.tail
                    
                    def search(self,object):
                        aux = self.head
                        while aux != None:
                            if str(object) == str(aux.getObject()):
                                return aux
                            aux = aux.Next
                        return None
                    
                    def add(self,object):
                        return self.insertTail(object)
                    
                    def insert(self, Ob, object):
                        try:
                            if type(Ob) == type(ListNode()) :
                                
                                if (Ob.Next == None) :
                                    self.add(object)
                                    
                                else:
                                    self.size += 1
                                    newNode = ListNode.NodeObject(object)
                                    newNode.Next = Ob.Next
                                    Ob.Next = newNode
                                return True
                            
                            else:
                                if Ob != None :
                                     node = self.search(Ob)
                                     if node != None :
                                         return self.insert(node, object)
                                     else: 
                                         return False
                                else:
                                    return False
                        except:
                            return False
                    
                    def insertHead(self, object):
                        try:
                            if self.isEmpty():
                                self.head = ListNode.NodeObject(object)
                                self.tail = self.head
                            else:
                                self.head = ListNode.NodeObjectNext(object, self.head)
                            
                            self.size += 1
                            return True
                        except:
                            return False
                    
                    def insertTail(self, object):
                        try:
                            if self.isEmpty():
                                self.head = ListNode.NodeObject(object)
                                self.tail = self.head
                            else:
                                self.tail.Next = ListNode.NodeObject(object)
                                self.tail = self.tail.Next
                            
                            self.size += 1
                            return True
                        except:
                            return False
                    
                    def remove(self, object):
                        try:
                            if type(ListNode()) == type(object):
                                self.remove(object.getObject())
                            else:
                                a = self.head
                                t = None
                                eliminado = False
                                while a != None and eliminado == False :
                                    if a.getObject() == object:
                                        self.size -= 1
                                        if t == None :
                                            self.head = a.Next
                                            a.Next = None
                                        else:
                                            t.Next = a.Next
                                            a.Next = None
                                            eliminado = True
                                    t = a
                                    a = a.Next
                                return eliminado
                        except:
                            return False
                    
                    def contains(self, object):
                        return self.search(object) != None
                    
                    def toArray(self):
                        if self.isEmpty():
                            return np.zeros(0)
                        else:
                            a = self.head
                            o = np.zeros(self.size)
                            for i in range(0,self.size):
                                if(a == None):
                                    break;
                                o[i] = a.getObject()
                                a = a.Next
                            return o

                        if type(node) == type(ListNode()):
                            try:
                                a = self.head
                                p = None
                                while a != node:
                                    p = a
                                    a = a.Next
                                return p
                            except:
                                return None
                        else:
                            return None
                    
                    def getNextTo(self):
                        return self.head.Next.getObject()
                    
                    def getNextToNode(self, node):
                        if type(node) == type(ListNode()):
                            try:
                                return node.Next.getObject()
                            except:
                                return None
                        else:
                            return None

                        try:
                            if self.isEmpty() == False:
                                subLista = List()

                                    while From != to:
                                        subLista.add(From.getObject())
                                        From = From.Next
                                    subLista.add(From.getObject())
                                return subLista
                        except:

                    def sortList(self):
                        if(self.isEmpty() == False):
                            arr = self.toArray()
                            self.head = None
                            for i in range(self.getSize()):
                                for j in range(self.getSize()-1):
                                    if arr[j] > arr[j+1]:
                                        temp = arr[j]
                                        arr[j] = arr[j+1]
                                        arr[j+1] = temp
                            for elemento in arr:
                                self.add(elemento)
                        else:
                            return None
                        
                    def __iter__(self):
                        self.inode = self.head
                        return self
                    
                    def __next__(self):
                        if self.inode == None:
                            raise StopIteration
                        else:
                            item = self.inode.getObject()
                            self.inode = self.inode.Next
                            return item

                        return str(self.head)
