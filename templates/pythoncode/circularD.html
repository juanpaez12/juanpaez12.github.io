                            return aux
                        aux = aux.Next
                    return None

                def remove(self, object):
                    try:
                        if type(ListNode()) == type(object):
                            self.remove(object.getObject())
                        else:
                            if object != None:
                                eliminado = False
                                node = self.head
                                s = 0;
                                while eliminado == False and s < self.size :
                                    if node.getObject() == object :
