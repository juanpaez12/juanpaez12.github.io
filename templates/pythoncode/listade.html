                # -*- coding: utf-8 -*-

                from ListInterface import ListInterface
                from ListNode import ListNode
                import numpy as np
                
                class DoubleList(ListInterface):
                    
                    inode = ListNode()
                    size = 0
                    head = ListNode()
                    tail = ListNode()
                    
                    
                    def __init__(self):
                        self.clear()
                        
                    def DoubleList(self,object):
                        self.add(object)
                    
                    def isEmpty(self):
                        return self.head == None
                    
                    def getSize(self):
                        return self.size
                    
                    def clear(self):
                        self.head = None
                        self.tail = None
                        self.size = None
                    
                    def getHead(self):
                        return self.head
                    
                    def getTail(self):
                        return self.tail
                    
                    def search(self,object):
                        aux = self.head
                        while aux != None:
                            if str(object) == str(aux.getObject()):
                                return aux
                            aux = aux.Next
                        return None
                    
                    def add(self,object):
                        return self.insertTail(object)
                    
                    def insert(self, Ob, object):
                        try:
                            if type(Ob) == type(ListNode()) :
                                
                                if (Ob.Next == None) :
                                    self.add(object)
                                    
                                else:
                                    self.size += 1
                                    newNode = ListNode.NodeObjNxtPre(object,Ob.Next,Ob)
                                    newNode.Next.Previous = newNode
                                    Ob.Next = newNode
                                return True
                            
                            else:
                                
                                if Ob != None :
                                     node = self.search(Ob)
                                     if node != None :
                                         return self.insert(node, object)
                                     else: 
                                         return False
                                else:
                                    return False
                                
                        except:
                            return False
                    
                        
                    def insertHead(self, object):
                        try:
                            if self.isEmpty():
                                self.head = ListNode.NodeObject(object)
                                self.tail = self.head
                            else:
                                
                                self.head = ListNode.NodeObjNxtPre(object, self.head, None)
                                self.head.Next.Previous = self.head
                            
                            self.size += 1
                            return True
                        except:
                            return False
                    
                    def insertTail(self, object):
                        try:
                            if self.isEmpty():
                                self.head = ListNode.NodeObject(object)
                                self.tail = self.head
                            else:
                                self.tail.Next = ListNode.NodeObjNxtPre(object, None, self.tail)
                                self.tail = self.tail.Next
                            
                            self.size += 1
                            return True
                        except:
                            return False
                    
                    def remove(self, object):
                        try:
                            if type(ListNode()) == type(object):
                                self.remove(object.getObject())
                            else:
                                if object != None and self.isEmpty() != True :
                                    eliminado = False
                                    node = self.head
                                    while eliminado == False and node != None :
                                        if node.getObject() == object :
                                            self.size -= 1
                                            eliminado = True
                                            if node.Previous == None :
                                                self.head = self.head.Next
                                                self.head.Previous = None
                                            else:
                                                node.Previous.Next = node.Next
                                        node = node.Next
                                    return eliminado
                                else:
                                    return False
                        except:
                            return False
                        
                    
                    def contains(self, object):
                        return self.search(object) != None
                    
                    def toArray(self):
                        if self.isEmpty():
                            return np.zeros(0)
                        else:
                            a = self.head
                            o = np.zeros(self.size)
                            for i in range(0,self.size):
                                if(a == None):
                                    break;
                                o[i] = a.getObject()
                                a = a.Next
                            return o
                    
                    def getBeforeTo(self):
                        try:
                            return self.head.Previous.getObject()
                        except:
                            return None
                    
                    def getBeforeToNode(self, node):
                        try:
                            return node.Previous
                        except:
                            return None
                    
                    def getNextTo(self):
                        return self.head.Next.getObject()
                    
                    def getNextToNode(self, node):
                        if type(node) == type(ListNode()):
                            try:
                                return node.Next.getObject()
                            except:
                                return None
                        else:
                            return None
                    
                    def subList(self, From, to):
                        try:
                            if self.isEmpty() == False:
                                subLista = DoubleList()
                                if self.contains(From.getObject()) and self.contains(to.getObject()):
                                    while From != to:
                                        subLista.add(From.getObject())
                                        From = From.Next
                                    subLista.add(From.getObject())
                                return subLista
                        except:
                            return None
                        
                    def sortList(self):
                        if(self.isEmpty() == False):
                            arr = self.toArray()
                            self.head = None
                            for i in range(self.getSize()):
                                for j in range(self.getSize()-1):
                                    if arr[j] > arr[j+1]:
                                        temp = arr[j]
                                        arr[j] = arr[j+1]
                                        arr[j+1] = temp
                            for elemento in arr:
                                self.add(elemento)
                        else:
                            return None
                        
                    def __iter__(self):
                        self.inode = self.head
                        return self
                    
                    def __next__(self):
                        if self.inode == None:
                            raise StopIteration
                        else:
                            item = self.inode.getObject()
                            self.inode = self.inode.Next
                            return item
                    
                    def __str__(self):
                        node = self.head
                        txt = ""
                        while node != None :
                            txt += "<-" + str(node.getObject()) + "->"
                            node = node.Next
                        return txt
